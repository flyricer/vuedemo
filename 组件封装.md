# v-model语法糖

#### 父组件
```
<six v-model="counerVal"></six>
```
#### 子组件
```
<button @click="changeValue(-1)">-1</button>
<span>{{currentVal}}</span>
<button @click="changeValue(1)">+1</button>

props: {
    value: {
      type: Number
    }
},
model: {
    prop: 'value',
    event: 'inputs'
},
data() {
    return {
        currentVal: this.value
    };
},
methods: {
    changeValue(val) {
        this.currentVal += parseInt(val);
        this.$emit("inputs", this.currentVal); // 事件名可随意取名，只需与model中保持一致即可
    }
}
```
